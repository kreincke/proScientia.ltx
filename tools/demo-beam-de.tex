\section{Zitatmarkierung}

\begin{frame}
  \frametitle{Verifikation Zitatmarkierung}
  \begin{itemize}

    \item Ein deutschsprachiges \enquote{Inline-Zitat mit \enquote{eingebettetem Zitat}}.
  
    \item Deutschsprachiger Satz \foreignquote{german}{with embedded foreign phrase}
    als eingebettetes Zitat. Erwartetes Resultat: Deutsche Anführungszeichen,
    weil Teil in einem deutschen Satz.

    \item Autonomes deutsches Zitat in einem insgesamt deutschen Text:
    \newline \newline
    \begin{quote}
      \textrm{\enquote{Dies ist ein deutschsprachiger Absatz, dessen Sätzte zusammen
      ein wörtliches Zitat bilden.}}
    \end{quote}
    erwartetet: Deutsche Anführungszeichen.
  
    \item Autonomes englisches Zitat in einem insgesamt deutschen Text:
    \newline \newline
    \begin{quote}
      \textrm{\foreignquote{english}{This shall be an English written paragraph containing
      a set of sentences which together build the quote.}}
    \end{quote}
    erwartetet: Englische Anführungszeichen, weil autonomer Absatz.
  \end{itemize}

\end{frame}
